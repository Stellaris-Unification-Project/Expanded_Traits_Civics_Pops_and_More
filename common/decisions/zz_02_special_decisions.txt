decision_prospect = {
	owned_planets_only = yes

	enactment_time = 180
	resources = {
		category = decisions
		cost = {
			influence = 25
			energy = 500
		}
	}

		is_artificial = no
		NOT = { is_planet_class = pc_machine }
		NOT = { is_planet_class = pc_hive }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_relic }
		owner = {
			has_active_tradition = tr_adaptability_finish

		NOT = { has_planet_flag = planet_prospected }

	allow = {
		NOT = { has_modifier = delicate_ecosystem }
		NOT = { has_modifier = eco_fragile_ecosystem }
	}
	
	effect = {
		custom_tooltip = decision_prospect_effects_short
		hidden_effect = {
			set_planet_flag = planet_prospected

			create_message = {
				type = MESSAGE_TYPE_PLANET_PROSPECTED
				localization = MESSAGE_PLANET_PROSPECTED
				days = 30
				target = this
				variable = {
					type = name
					localization = PLANET
					scope = this
				}				
				variable = {
					type = name
					localization = DEPOSIT
					scope = last_added_deposit
				}
			}
		}
	}
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			num_pops < 20
		}
	}	
}

# Penal Colony
decision_penal_colony = {
	owned_planets_only = yes
	
	prerequisites = {
		"tech_penal_colonies"		
	}

	resources = {
		category = decisions
		cost = {
			influence = 100
		}
	}
	
	potential = {
		is_capital = no
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = eco_trash_colony }
		owner = { 
			is_regular_empire = yes 
		}
		is_artificial = no
		NOT = { 
			owner = { 
				any_owned_planet = { has_modifier = penal_colony }
			} 
		}
	}
	
	allow = {
		planet_size >= 15
		NOT = { has_modifier = delicate_ecosystem }
		NOT = { has_modifier = eco_fragile_ecosystem }
	}
	
	effect = {
		add_modifier = {
			modifier = penal_colony
			days = -1
		}
		hidden_effect = { validate_planet_buildings_and_districts = yes }
	}
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			NOR = {
				num_pops >= 25
				planet_crime >= 50
				owner = {
					count_owned_planet = {
						count > 3
						limit = { planet_crime >= 25 }
					}
				}
			}
		}
		modifier = {
			factor = 0
			OR = {
				owner = { num_owned_planets < 5 }
				is_planet_class = pc_gaia
				is_planet_class = pc_relic
				is_planet_class = pc_city
				habitability = { who = owner_species value > 0.5 }
				pop_percentage = {
					percentage > 0.25
					limit = {
						OR = {
							has_job_category = specialist
							has_job_category = ruler
						}
					}
				}
			}
		}
		modifier = {
			factor = 0
			owner = {
				OR = {
					is_egalitarian = yes
					is_civic_has_meritocracy = yes
				}
			}
		}
		modifier = {
			factor = 3
			owner = {
				OR = {
					is_authoritarian = yes
					is_xenophobe = yes
				}
			}
		}
	}
}

# Resort Colony
decision_resort_colony = {
	owned_planets_only = yes
	
	prerequisites = {
		"tech_resort_colonies"
	}

	resources = {
		category = decisions
		cost = {
			influence = 100
		}
	}
	
	potential = {
		is_capital = no
		NOT = { has_modifier = penal_colony }
		NOT = { has_modifier = slave_colony }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_relic }
		NOT = { is_planet_class = pc_nuked }
		NOT = { has_modifier = eco_trash_colony }
		owner = { 
			OR = {
				is_regular_empire = yes 
				has_valid_civic = civic_machine_servitor
			}
		}
		is_artificial = no
	}
	
	allow = {
		planet_size >= 15
		NOT = { has_modifier = delicate_ecosystem }
		NOT = { has_modifier = eco_fragile_ecosystem }
		NOT = { has_deposit = d_exploited_homeworld }
		custom_tooltip = {
			fail_text = decision_resort_colony_no_districts
			num_districts = { type = any value < 1 }
			num_buildings = { type = any value < 2 }
		}
		custom_tooltip = {
			fail_text = no_upgraded_capital_tooltip
			has_upgraded_capital = no
		}
		custom_tooltip = {
			fail_text = decision_resort_colony_only_one
			NOT = {
				owner = {
					any_owned_planet = { has_modifier = resort_colony }
				}
			}
		}
	}
	
	effect = {
		add_modifier = {
			modifier = resort_colony
			days = -1
		}
		if = {
			limit = { owner = { is_regular_empire = yes } }
			add_deposit = d_resort_world
		}
		else_if = {
			limit = { owner = { has_valid_civic = civic_machine_servitor } }
			add_deposit = d_resort_world_robot
		}
		hidden_effect = {
			remove_building = building_colony_shelter
			add_building = building_resort_capital
		}
		hidden_effect = { validate_planet_buildings_and_districts = yes }
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 2
			exists = owner
			owner = {
				has_monthly_income = { resource = energy value > 25 }
				has_monthly_income = { resource = minerals value > 25 }
				OR = {
					has_monthly_income = { resource = food value > 25 }
					country_uses_food = no
				}
				has_monthly_income = { resource = alloys value > 50 }
				has_monthly_income = { resource = consumer_goods value > 25 }
			}
			NOR = {
				any_deposit = { has_deposit_category = deposit_cat_rare }
				AND = {
					num_free_districts = { type = district_farming value > 5 }
					country_uses_food = yes
				}
				num_free_districts = { type = district_mining value > 5 }
				num_free_districts = { type = district_generator value > 5 }
			}
		}
		modifier = {
			factor = 0
			OR = {
				owner = { num_owned_planets < 5 }
				is_planet_class = pc_nuked
				has_modifier = hazardous_weather
				has_modifier = dangerous_wildlife
				has_modifier = unstable_tectonics
				has_modifier = asteroid_impacts
				has_modifier = wild_storms
				has_modifier = mineral_rich
				has_modifier = ultra_rich
				has_modifier = irradiated_planet
			}
		}
		modifier = {
			factor = 2
			OR = {
				has_modifier = titanic_life
				has_modifier = natural_beauty
				has_modifier = atmospheric_aphrodisiac
				has_modifier = atmospheric_hallucinogen
				has_modifier = lush_planet
			}
		}
		modifier = {
			factor = 3
			is_planet_class = pc_gaia
		}
	}
}

							pop = this

					}
				}
			}
			else = {
				add_deposit = d_lithoid_devastation
				add_planet_devastation = 10

				days = 30
				target = this

					type = name

					scope = this

				limit = {
					num_free_districts = { type = any value <= 4 }

				}
				random_deposit = {
					limit = {
						is_blocker = yes
					}
					clear_blocker = yes
				}
			}

			owner = { country_event = { id = colony.185 days = 1 } }
		}

			is_lithoid_devouring_swarm_consumption_world = no
		}

	owned_planets_only = yes
	icon = decision_arcology_project

	enactment_time = 3600
	resources = {
		category = decisions

			has_technology = tech_starbase_5

			has_technology = tech_zero_point_power
		}
	}

		# Make ringworld
		custom_tooltip = decision_shattered_ring_project_effects
		hidden_effect = {
			clear_deposits = yes
			change_pc = pc_ringworld_habitable

			if = {
				limit = { has_building = building_mineral_purification_plant }
				remove_building = building_mineral_purification_plant
			}

			if = {
				limit = { has_building = building_mineral_purification_hub }
				remove_building = building_mineral_purification_hub
			}

			# This removes the extra districts when fixing the ring world.
			if = {
				limit = {
					owner = { is_regular_empire = yes }
				}
				shattered_ring_world_conversion_effect = {
					DISTRICT = district_rw_city
					VARIABLE = num_housing_districts
				}
				shattered_ring_world_conversion_effect = {
					DISTRICT = district_rw_commercial
					VARIABLE = num_commercial_districts
				}
			}

			if = {
				limit = {
					owner = { is_hive_empire = yes }
				}
				shattered_ring_world_conversion_effect = {
					DISTRICT = district_rw_hive
					VARIABLE = num_housing_districts
				}
				shattered_ring_world_conversion_effect = {
					DISTRICT = district_rw_generator
					VARIABLE = num_generator_districts
				}
			}

				limit = {

					VARIABLE = num_housing_districts

					DISTRICT = district_rw_generator
					VARIABLE = num_generator_districts
				}
			}

				DISTRICT = district_rw_farming
				VARIABLE = num_farming_districts
			}
			shattered_ring_world_conversion_effect = {
				DISTRICT = district_rw_industrial
				VARIABLE = num_industrial_districts

			factor = 0
			exists = owner
